246 void Graph::articulationDFS(Graph::Node &node, std::vector<Airport*> &points, std::stack<Node> &st, int index) {
247  node.visited = true;
248  node.num = index;
249  node.low = index;
250  st.push(node);
251  node.inStack = true;
252  index++;
253  bool flag = false;
254  int child = 0;
255 
256  for (Edge& edge : node.adj) {
257  auto& destNode = nodes[edge.destination];
258  if(!destNode.visited) {
259  child++;
260  articulationDFS(destNode, points, st, index);
261  node.low = std::min(node.low, destNode.low);
262  if (destNode.low >= node.num) flag = true; // it is an art. point
263  }
264  else if (destNode.inStack) {
265  node.low = std::min(node.low, destNode.num);
266  }
267  }
268 
269  // articulation point if and only if it has more than one child on the dfs tree
270 
271  if ((node.num == node.low && child > 1) || (node.num != node.low && flag)) {
272  while (!st.empty()) {
273  Node top = st.top();
274  st.pop();
275  node.inStack = false;
276 
277  if (top.airport->getName() == node.airport->getName()) {
278  points.push_back(top.airport);
279  break;
280  }
281  }
282  }
